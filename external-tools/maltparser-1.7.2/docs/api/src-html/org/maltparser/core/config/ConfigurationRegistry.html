001    package org.maltparser.core.config;
002    
003    import java.util.HashMap;
004    /**
005    *
006    * @author Johan Hall
007    **/
008    public class ConfigurationRegistry extends HashMap<Class<?>, Object> {
009            public final static long serialVersionUID = 3256444702936019250L;
010            
011            public ConfigurationRegistry() {
012                    super();
013            }
014    
015            @Override
016            public Object get(Object key) {
017                    return super.get(key);
018            }
019    
020            @Override
021            public Object put(Class<?> key, Object value) {
022                    return super.put(key, value);
023            }
024    }
