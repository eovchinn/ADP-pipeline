051            public PlanarConfig(SymbolTableHandler symbolTableHandler, String noCoveredRoots , String acyclicity , String connectedness , String rootHandling ) throws MaltChainedException {
052                    super();
053                    stack = new Stack<DependencyNode>();
054                    input = new Stack<DependencyNode>();
055                    dependencyGraph = new DependencyGraph(symbolTableHandler);
056                    setRootHandling(rootHandling);
057                    setNoCoveredRoots(Boolean.valueOf(noCoveredRoots));
058                    setAcyclicity(Boolean.valueOf(acyclicity));
059                    setConnectedness(connectedness);
060            }

077            
078            public DependencyNode getStackNode(int index) throws MaltChainedException {
079                    if (index < 0) {
080                            throw new ParsingException("Stack index must be non-negative in feature specification. ");
081                    }
082                    if (stack.size()-index > 0) {
083                            return stack.get(stack.size()-1-index);
084                    }
085                    return null;
086            }
087            
088            public DependencyNode getInputNode(int index) throws MaltChainedException {
089                    if (index < 0) {
090                            throw new ParsingException("Input index must be non-negative in feature specification. ");
091                    }
092                    if (input.size()-index > 0) {
093                            return input.get(input.size()-1-index);
094                    }       
095                    return null;
096            }
097            
098            public void setDependencyGraph(DependencyStructure source) throws MaltChainedException {
099                    dependencyGraph.clear();
100                    for (int index : source.getTokenIndices()) {
101                            DependencyNode gnode = source.getTokenNode(index);
102                            DependencyNode pnode = dependencyGraph.addTokenNode(gnode.getIndex());
103                            for (SymbolTable table : gnode.getLabelTypes()) {
104                                    pnode.addLabel(table, gnode.getLabelSymbol(table));
105                            }
106                            
107                            if (gnode.hasHead()) {
108                                    Edge s = gnode.getHeadEdge();
109                                    Edge t = dependencyGraph.addDependencyEdge(s.getSource().getIndex(), s.getTarget().getIndex());
110                                    
111                                    for (SymbolTable table : s.getLabelTypes()) {
112                                            t.addLabel(table, s.getLabelSymbol(table));
113                                    }
114                            }
115                    }
116                    for (SymbolTable table : source.getDefaultRootEdgeLabels().keySet()) {
117                            dependencyGraph.setDefaultRootEdgeLabel(table, source.getDefaultRootEdgeLabelSymbol(table));
118                    }
119            }
120            
121            public DependencyStructure getDependencyGraph() {
122                    return dependencyGraph;
123            }
124            
125            public void initialize(ParserConfiguration parserConfiguration) throws MaltChainedException {
126                    if (parserConfiguration != null) {
127                            PlanarConfig planarConfig = (PlanarConfig)parserConfiguration;
128                            Stack<DependencyNode> sourceStack = planarConfig.getStack();
129                            Stack<DependencyNode> sourceInput = planarConfig.getInput();
130                            setDependencyGraph(planarConfig.getDependencyGraph());
131                            for (int i = 0, n = sourceStack.size(); i < n; i++) {
132                                    stack.add(dependencyGraph.getDependencyNode(sourceStack.get(i).getIndex()));
133                            }
134                            for (int i = 0, n = sourceInput.size(); i < n; i++) {
135                                    input.add(dependencyGraph.getDependencyNode(sourceInput.get(i).getIndex()));
136                            }
137                    } else {
138                            stack.push(dependencyGraph.getDependencyRoot());
139                            for (int i = dependencyGraph.getHighestTokenIndex(); i > 0; i--) {
140                                    final DependencyNode node = dependencyGraph.getDependencyNode(i);
141                                    if (node != null && !node.hasHead()) { 
142                                            input.push(node);
143                                    }
144                            }
145                    }
146            }

183            protected void setConnectedness(String conn) throws MaltChainedException {
184                    if (conn.equalsIgnoreCase("none")) {
185                            connectedness = NO_CONNECTEDNESS;
186                    } else if (conn.equalsIgnoreCase("reduceonly")) {
187                            connectedness = REDUCE_ONLY;
188                    } else if (conn.equalsIgnoreCase("full")) {
189                            connectedness = FULL_CONNECTEDNESS;
190                    } else {
191                            throw new ParsingException("The connectedness constraint option '"+conn+"' is unknown");
192                    }
193            }

234            public boolean equals(Object obj) {
235                    if (this == obj)
236                            return true;
237                    if (obj == null)
238                            return false;
239                    if (getClass() != obj.getClass())
240                            return false;
241                    PlanarConfig that = (PlanarConfig)obj;
242                    
243                    if (stack.size() != that.getStack().size()) 
244                            return false;
245                    if (input.size() != that.getInput().size())
246                            return false;
247                    if (dependencyGraph.nEdges() != that.getDependencyGraph().nEdges())
248                            return false;
249                    for (int i = 0; i < stack.size(); i++) {
250                            if (stack.get(i).getIndex() != that.getStack().get(i).getIndex()) {
251                                    return false;
252                            }
253                    }
254                    for (int i = 0; i < input.size(); i++) {
255                            if (input.get(i).getIndex() != that.getInput().get(i).getIndex()) {
256                                    return false;
257                            }
258                    }               
259                    return dependencyGraph.getEdges().equals(that.getDependencyGraph().getEdges());
260            }

267                    sb.append(", ");
268                    sb.append(dependencyGraph.nEdges());
269                    return sb.toString();
270            }
